select concat(FName,'  ', LName) AS `NAME`,
concat (LPAD(C.ClientIDBrCode,4, '0'), '-',
      LPAD(C.ClientID,7, '0'), '-',
      LPAD(C.ClientIDChkDgt,1, '0')) AS `CLIENT ID`,

CL.ClientTypeDesc as `CLIENT TYPE`,
G.GeoDesc as `GEOGRAPHICAL`,
AT.AcctTypeDesc as `ACCOUNT TYPE`,
AL.AlertLevelDesc as `ALERT LEVEL`,
AC.AcctStatDesc as `ACCOUNT STATUS`,
CV.CivilStatDesc as `CIVIL STATUS`,

(SELECT SUM(AMT*-1) FROM SLDTL AS SL WHERE SLC_CODE=31 AND SLT_CODE=1 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `SHARE CAP`,

(SELECT SUM(AMT*-1) FROM SLDTL AS  SL WHERE SLC_CODE=22 AND SLC_CODE=1 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `SAVINGS`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=1 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `EMERGENCY LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=2 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `KAUNLARAN LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=3 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `PRODUCTION LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=4 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `PUHUNAN LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=5 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `PENSIONER LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=6 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `MICRO LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=7 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `SALARY LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=8 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `EMERGENCY LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=9 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `HOG RAISING LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=10 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `REGULAR LOAN`,

(SELECT SUM(AMT*1) FROM SLDTL AS SL WHERE SLC_CODE=12 AND SLT_CODE=11 AND SLE_CODE=11
AND SL.SL_BRCODE=C.CLIENTIDBRCODE
AND SL.SL_CLIENTID=C.CLIENTID) AS `SPECIAL LOAN`
from client as C

LEFT JOIN clienttype as CL ON CL.ClientTypeID=C.CLIENTTYPE
LEFT JOIN geo as G ON G.GeoID=C.GEOGRAPHICAL
LEFT JOIN accttype as AT ON AT.AccTypeID=C.ACCOUNTTYPE
LEFT JOIN alertlevel as AL ON AL.AlertLevelID=C.ALERTLEVEL
LEFT JOIN acctstat as AC ON AC.AcctStatID=C.ACCOUNTSTATUS
LEFT JOIN civilstat as CV ON CV.CivilStatID=C.CIVILSTATUS
GROUP BY CLIENTID;
